function loadInventory() { if (lineName === 'Unassigned') { loadOrphans(); return; } fetch('/api/inventory').then(function(r){ return r.json(); }).then(function(d) { if (!d.success) { document.getElementById('inv-loading').textContent = 'Error: '+(d.error||'unknown'); return; } window._locations = d.locations || []; var inv = d.inventory || {}; var lineUpper = lineName.toUpperCase(); // For Apollo, stations are the group keys — collect all station groups // For other lines, there's one group matching the line name var isApollo = lineUpper === 'APOLLO'; var lineGroups; if (isApollo) { // Apollo station groups: any group whose items have line === 'APOLLO' lineGroups = Object.keys(inv).filter(function(grp) { var items = inv[grp] || []; return items.length > 0 && String(items[0].line || '').toUpperCase() === 'APOLLO'; }); } else { lineGroups = Object.keys(inv).filter(function(grp) { return grp.toUpperCase() === lineUpper; }); } // Collect all in-stock parts for this line var inStock = {}; var inStockItems = []; lineGroups.forEach(function(grp) { (inv[grp] || []).forEach(function(item) { var key = item.partNum.toLowerCase(); if (inStock[key]) return; inStock[key] = true; inStockItems.push(item); }); }); // Now find BOM parts for this line that have no locations — mark as out of stock // Use the same line groups to find what parts belong to this line in the BOM var bomForLine = {}; lineGroups.forEach(function(grp) { (inv[grp] || []).forEach(function(item) { var key = item.partNum.toLowerCase(); if (!bomForLine[key]) bomForLine[key] = { partNum: item.partNum, partName: item.partName }; }); }); // Also check full BOM list for parts on this line that have zero stock // The BOM list has all parts; cross-ref with locations to find zero-stock ones var bomList = d.bomList || []; var locsByPart = {}; (d.locations || []).forEach(function(l) { var key = l.partNum.toLowerCase(); if (!locsByPart[key]) locsByPart[key] = 0; locsByPart[key] += parseInt(l.quantity) || 0; }); // Build final allItems — in-stock items first, then out-of-stock BOM parts allItems = inStockItems.slice(); Object.keys(bomForLine).forEach(function(pnKey) { if (inStock[pnKey]) return; // already in list with locations // This part is on the BOM for this line but has zero or no locations var p = bomForLine[pnKey]; allItems.push({ partNum: p.partNum, partName: p.partName, outOfStock: true }); }); document.getElementById('inv-loading').style.display = 'none'; document.getElementById('inv-form').style.display = ''; renderParts(''); }).catch(function(e){ document.getElementById('inv-loading').textContent = 'Failed: '+e.message; document.getElementById('inv-form').style.display = ''; }); }
