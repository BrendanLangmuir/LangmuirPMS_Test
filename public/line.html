function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); // ── Read Locations tab ─────────────────────────────────── const locSheet = ss.getSheetByName('Locations'); if (!locSheet) throw new Error('Locations sheet not found'); const locRows = locSheet.getDataRange().getValues(); // Real data in columns Q(16), R(17), S(18), T(19) // Headers: Location | Qty | Part Number | Description const locMap = {}; const locations = []; for (let i = 1; i < locRows.length; i++) { const location = String(locRows[i][16] || '').trim(); // Q const quantity = String(locRows[i][17] || '').trim(); // R const partNum = String(locRows[i][18] || '').trim(); // S const partName = String(locRows[i][19] || '').trim(); // T if (!partNum) continue; const entry = { location, partNum, partName, quantity }; locations.push(entry); const key = partNum.toLowerCase(); if (!locMap[key]) locMap[key] = []; locMap[key].push(entry); } // ── Read BOM tab ───────────────────────────────────────── const bomSheet = ss.getSheetByName('BOM'); if (!bomSheet) throw new Error('BOM sheet not found'); const bomRows = bomSheet.getDataRange().getValues(); // Headers: Parent Part | Station | Part | Description const inventory = {}; // grouped by station (Apollo) or line (others) for (let i = 1; i < bomRows.length; i++) { const [parentPart, station, partNum, description] = bomRows[i]; if (!partNum || String(partNum).trim() === 'Hardware') continue; const line = String(parentPart).trim().toUpperCase(); const st = String(station || '').trim(); const pn = String(partNum ).trim(); const desc = String(description || '').trim(); // Only include parts that exist in Locations tab — match by part number only const locEntries = locMap[pn.toLowerCase()]; if (!locEntries || locEntries.length === 0) continue; const primaryLoc = locEntries[0]; const item = { partNum: pn, partName: primaryLoc.partName || desc, // use Locations description, fall back to BOM line, station: st, location: primaryLoc.location, quantity: primaryLoc.quantity, allLocations: locEntries.map(l => ({ location: l.location, quantity: l.quantity })), }; const groupKey = line === 'APOLLO' ? (st || 'APOLLO') : line; if (!inventory[groupKey]) inventory[groupKey] = []; // Only add if part number not already in this group const already = inventory[groupKey].some(i => i.partNum.toLowerCase() === pn.toLowerCase()); if (!already) inventory[groupKey].push(item); } // Sort each group alphabetically by part number Object.keys(inventory).forEach(key => { inventory[key].sort((a, b) => a.partNum.localeCompare(b.partNum, undefined, { numeric: true, sensitivity: 'base' })); }); return ContentService .createTextOutput(JSON.stringify({ success: true, locations, inventory })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { const data = JSON.parse(e.postData.contents); if (data.action === 'subtract') { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('Locations'); if (!sheet) throw new Error('Locations sheet not found'); const rows = sheet.getDataRange().getValues(); const partNum = String(data.partNum || '').trim().toLowerCase(); const partName = String(data.partName || '').trim().toLowerCase(); const qty = parseInt(data.qty) || 1; for (let i = 1; i < rows.length; i++) { const rowPn = String(rows[i][18] || '').trim().toLowerCase(); // col S = Part Number const rowPna = String(rows[i][19] || '').trim().toLowerCase(); // col T = Description if ((partNum && rowPn === partNum) || (!partNum && partName && rowPna === partName)) { const currentQty = parseInt(rows[i][17]) || 0; // col R = Qty const newQty = Math.max(0, currentQty - qty); sheet.getRange(i + 1, 18).setValue(newQty); // col R = column 18 return ContentService .createTextOutput(JSON.stringify({ success: true, newQty })) .setMimeType(ContentService.MimeType.JSON); } } return ContentService .createTextOutput(JSON.stringify({ success: false, error: 'Part not found in Locations' })) .setMimeType(ContentService.MimeType.JSON); } return ContentService .createTextOutput(JSON.stringify({ success: false, error: 'Unknown action' })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } }
